Question 2.

Count the Heads

Solution - 

        var Count_Occ = (s) => {
            let arr = s.split("").sort();
            let heads = "";
            let visit = [];
            let k = 0;
            let res = "";
            for(let i=0;i < arr.length;i++)
            {
              let count = 0;
              for(let j=i;j < arr.length;j++)
              {
                if(arr[i] === arr[j])
                {
                  heads = arr[i];
                  count++;
                }
              }
              if(!visit.includes(heads) && count >= 2)
              {
                visit[k] = heads;
                visit[k+1] = count;
                k= k + 2;
              }
            }
              res = visit.join("");
            return res;
        };

Question 6.

Find the Winner

Solution - 

        var Game_Winner = (S) => 
        {
          let Aletter=0,Dletter=0;
          for(let i=0;i < S.length;i++){
            if(S[i]==="A")
              Aletter+=1;
            else if(S[i]==="D")
            Dletter+=1;
          }
          if(Aletter < Dletter)
          return "Danish";
          else if(Aletter > Dletter)
          return "Aditya";
          else if(Aletter === Dletter)
          return "Draw";
        };

Question 8.

Plaindrome Check

Solution - 

        var Palin_Check = (str) =>
      {
         //a number is same after reverse is known as palindrome.
         let reverseStr="";
         for(let i = str.length-1;i>=0;i--){
           reverseStr+=str[i];
         }
         if(reverseStr===str)
         return "True";
         else
         return "False";
      }

Solution - 

        var Count_Vowels= (S) => 
      {
        var count=0;
        // let vowels=`aeiou`
        for(let i=0;i < S.length;i++){
          var letter=S[i].toLowerCase()
          if(letter==="a" || letter==="e" || letter==="i" ||
          letter==="o" || letter==="u"){
            count++;
          }
        }
        return count;
            
      };
      var Count_Consonants= (S) => 
      {
        counter=0;
          for(let i=0;i < S.length;i++){
          var letter=S[i].toLowerCase()
          if(letter!=="a" && letter!=="e" && letter!=="i" &&
          letter!=="o" && letter!=="u"){
            counter++;
          }
            
          }

      };
